    <script>
        async function summarize() {
            const key = document.getElementById('apiKey').value;
            const text = document.getElementById('inputText').value;
            if(!key || !text) { alert('키와 내용을 모두 입력해주세요!'); return; }

            const loading = document.getElementById('loading');
            const resultDiv = document.getElementById('result');
            
            loading.style.display = 'block';
            resultDiv.style.display = 'none';

            const prompt = `너는 해외영업팀의 유능한 전략 기획자야. 다음의 중구난방인 회의록이나 메모를 분석해서 딱딱한 비즈니스 보고서 형식으로 요약해줘. 
            반드시 다음 세 가지 체계를 지켜서 작성할 것:
            1. [현황]: 현재 진행 중인 상황 요약
            2. [이슈]: 발생한 문제점이나 리스크
            3. [대응 및 Action Item]: 향후 계획과 담당자가 해야 할 일
            
            내용: ${text}`;

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${key}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] })
                });

                const data = await response.json();
                const output = data.candidates[0].content.parts[0].text;
                
                resultDiv.innerText = output;
                resultDiv.style.display = 'block';
            } catch (e) {
                alert('에러 발생: API 키를 확인해주세요.');
            } finally {
                loading.style.display = 'none';
            }
        }
    </script>
